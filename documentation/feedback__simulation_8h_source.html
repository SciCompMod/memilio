212  auto advance(const FP tmax, const FP dt_feedback = 1.0)
213  {
214  FP t = m_simulation.get_result().get_last_time();
215  while (t < tmax) {
216  FP dt_eff = std::min(dt_feedback, tmax - t);
217  apply_feedback(t + dt_eff);
218  m_simulation.advance(t + dt_eff);
219  add_icu_occupancy(t + dt_eff);
220  t += dt_eff;
221  }
222  return m_simulation.get_result().get_last_value();
223  }

269  FP calc_risk_perceived()
270  {
271  const auto& icu_occ = m_feedback_parameters.template get<ICUOccupancyHistory<FP>>();
272  size_t num_time_points = icu_occ.get_num_time_points();
273  size_t n = std::min(static_cast<size_t>(num_time_points), m_feedback_parameters.template get<GammaCutOff>());
274  FP perceived_risk = 0.0;
275  const auto& a = m_feedback_parameters.template get<GammaShapeParameter<FP>>();
276  const auto& b = m_feedback_parameters.template get<GammaScaleParameter<FP>>();
277  for (size_t i = num_time_points - n; i < num_time_points; ++i) {
278  size_t day = i - (num_time_points - n);
279  FP gamma = std::pow(b, a) * std::pow(day, a - 1) * std::exp(-b * day) / std::tgamma(a);
280  FP perc_risk = icu_occ.get_value(i).sum() / m_feedback_parameters.template get<NominalICUCapacity<FP>>();
281  perc_risk = std::min(perc_risk, 1.0);
282  perceived_risk += perc_risk * gamma;
283  }
284  return perceived_risk;
285  }

296  void add_icu_occupancy(FP t)
297  {
298  auto& model = m_simulation.get_model();
299  size_t num_groups = static_cast<size_t>(model.parameters.get_num_groups());
300  const auto& last_values = m_simulation.get_result().get_last_value();
301  Eigen::VectorXd icu_occ(num_groups);
302  for (size_t i = 0; i < m_icu_indices.size(); ++i) {
303  icu_occ[i] = last_values[m_icu_indices[i]];
304  }
305  // normalize by total population and scale to 100,000 inhabitants
306  icu_occ = icu_occ / model.populations.get_total() * 100000;
307  m_feedback_parameters.template get<ICUOccupancyHistory<FP>>().add_time_point(t, icu_occ);
308  }

318  void apply_feedback(FP t)
319  {
320  // get model parameters
321  auto& params = m_simulation.get_model().parameters;
322  const auto num_groups = (size_t)params.get_num_groups();
323 
324  // get feedback parameters
325  const auto& contactReductionMax = m_feedback_parameters.template get<ContactReductionMax<FP>>();
326  const auto& contactReductionMin = m_feedback_parameters.template get<ContactReductionMin<FP>>();
327  const auto epsilon = m_feedback_parameters.template get<SoftPlusCurvatureParameter<FP>>();
328 
329  const size_t num_locations = contactReductionMax.size();
330 
331  // calculate local perceived risk
332  FP perceived_risk = calc_risk_perceived();
333 
334  // add the perceived risk to the time series
335  m_perceived_risk.add_time_point(t, Eigen::VectorXd::Constant(num_groups, perceived_risk));
336 
337  auto group_weights_all = Eigen::VectorXd::Constant(num_groups, 1.0);
338 
339  // define a lambda function to create a damping sampling object given a reduction factor and location
340  auto set_contact_reduction = [=](FP val, size_t location) {
341  auto v = mio::UncertainValue(val);
342  return mio::DampingSampling(v, mio::DampingLevel(0), mio::DampingType(0), mio::SimulationTime(t),
343  std::vector<size_t>{location}, group_weights_all);
344  };
345 
346  auto& dampings = params.template get<ContactPatterns>().get_dampings();
347 
348  // for each location, compute the effective contact reduction factor and add the corresponding damping sample
349  for (size_t loc = 0; loc < num_locations; ++loc) {
350 
351  // compute the effective reduction factor using a softplus function.
352  ScalarType reduc_fac = (contactReductionMax[loc] - contactReductionMin[loc]) * epsilon *
353  std::log(std::exp(perceived_risk / epsilon) + 1.0) +
354  contactReductionMin[loc];
355 
356  // clamp the reduction factor to the maximum allowed value.
357  reduc_fac = std::min(reduc_fac, contactReductionMax[loc]);
358 
359  // generate the damping for the current location.
360  auto damping = set_contact_reduction(reduc_fac, loc);
361 
362  // add the damping to the contact pattern.
363  dampings.push_back(damping);
364  }
365  // update the contact matrices after all dampings have been added.
366  params.template get<ContactPatterns>().make_matrix();
367  }
