61  ScalarType evaluate(const AdoptionRate<Status>& rate, const Eigen::VectorXd& x) const
62  {
63  const auto& pop = this->populations;
64  const auto source = pop.get_flat_index({rate.region, rate.from});
65  // determine order and calculate rate
66  if (rate.influences.size() == 0) { // first order adoption
67  return rate.factor * x[source];
68  }
69  else { // second order adoption
70  ScalarType N = 0;
71  for (size_t s = 0; s < static_cast<size_t>(Status::Count); ++s) {
72  N += x[pop.get_flat_index({rate.region, Status(s)})];
73  }
74  // accumulate influences
75  ScalarType influences = 0.0;
76  for (size_t i = 0; i < rate.influences.size(); i++) {
77  influences +=
78  rate.influences[i].factor * x[pop.get_flat_index({rate.region, rate.influences[i].status})];
79  }
80  return (N > 0) ? (rate.factor * x[source] * influences / N) : 0;
81  }
82  }
