179  template <class IOContext>
180  static IOResult<UncertainValue> deserialize(IOContext& io)
181  {
182  auto obj = io.expect_object("UncertainValue");
183  if (!(io.flags() & IOF_OmitValues) && !(io.flags() & IOF_OmitDistributions)) {
184  auto v = obj.expect_element("Value", Tag<FP>{});
185  auto d = obj.expect_optional("Distribution", Tag<std::shared_ptr<ParameterDistribution>>{});
186  return apply(
187  io,
188  [](auto&& v_, auto&& d_) {
189  auto uv = UncertainValue(v_);
190  if (d_) {
191  uv.set_distribution(**d_);
192  }
193  return uv;
194  },
195  v, d);
196  }
197  else if (!(io.flags() & IOF_OmitValues) && (io.flags() & IOF_OmitDistributions)) {
198  auto v = obj.expect_element("Value", Tag<FP>{});
199  return apply(
200  io,
201  [](auto&& v_) {
202  return UncertainValue(v_);
203  },
204  v);
205  }
206  else if ((io.flags() & IOF_OmitValues) && !(io.flags() & IOF_OmitDistributions)) {
207  auto d = obj.expect_optional("Distribution", Tag<std::shared_ptr<ParameterDistribution>>{});
208  return apply(
209  io,

211  auto uv = UncertainValue();
212  if (d_) {
213  uv.set_distribution(**d_);
214  }
215  return uv;
216  },
217  d);
218  }
219  else {
220  return failure(StatusCode::InvalidValue,
