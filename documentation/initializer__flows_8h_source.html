134  template <size_t Group = 0>
135  bool compute_initialization_vector_impl(Eigen::VectorX<ScalarType>& init,
136  Eigen::VectorX<ScalarType> const& total_population,
137  Eigen::VectorX<ScalarType> const& deaths,
138  Eigen::VectorX<ScalarType> const& total_confirmed_cases)
139  {
140  static_assert((Group < Model::num_groups) && (Group >= 0), "The template parameter Group should be valid.");
141  using LctStateGroup = type_at_index_t<Group, LctStatesGroups>;
142  Eigen::Index first_index = m_model.populations.template get_first_index_of_group<Group>();
143  Eigen::Index first_index_flows = Group * (Eigen::Index)InfectionTransition::Count;
144 
145  bool error =
146  // Exposed.
147  (compute_compartment<InfectionState::Exposed, Group>(
148  init, first_index_flows + (Eigen::Index)InfectionTransition::SusceptibleToExposed,
149  1 / m_model.parameters.template get<TimeExposed>()[Group])) ||
150  // InfectedNoSymptoms.
151  (compute_compartment<InfectionState::InfectedNoSymptoms, Group>(
152  init, first_index_flows + (Eigen::Index)InfectionTransition::ExposedToInfectedNoSymptoms,
153  1 / m_model.parameters.template get<TimeInfectedNoSymptoms>()[Group])) ||
154  // InfectedSymptoms.
155  (compute_compartment<InfectionState::InfectedSymptoms, Group>(
156  init, first_index_flows + (Eigen::Index)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms,
157  1 / m_model.parameters.template get<TimeInfectedSymptoms>()[Group])) ||
158  // InfectedSevere.
159  (compute_compartment<InfectionState::InfectedSevere, Group>(
160  init, first_index_flows + (Eigen::Index)InfectionTransition::InfectedSymptomsToInfectedSevere,
161  1 / m_model.parameters.template get<TimeInfectedSevere>()[Group])) ||
162  // InfectedCritical.
163  (compute_compartment<InfectionState::InfectedCritical, Group>(
164  init, first_index_flows + (Eigen::Index)InfectionTransition::InfectedSevereToInfectedCritical,
165  1 / m_model.parameters.template get<TimeInfectedCritical>()[Group]));
166  if (error) {
167  return true;
168  }
169 
170  // Recovered.
171  // Number of recovered is equal to the cumulative number of confirmed cases minus the number of people
172  // in the group who are infected at the moment.
173  init[first_index + LctStateGroup::template get_first_index<InfectionState::Recovered>()] =
174  total_confirmed_cases[Group] -
175  init.segment(first_index + LctStateGroup::template get_first_index<InfectionState::InfectedSymptoms>(),
176  LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSymptoms>() +
177  LctStateGroup::template get_num_subcompartments<InfectionState::InfectedSevere>() +
178  LctStateGroup::template get_num_subcompartments<InfectionState::InfectedCritical>())
179  .sum() -
180  deaths[Group];
181 
182  // Susceptibles.
183  init[first_index + LctStateGroup::template get_first_index<InfectionState::Susceptible>()] =
184  total_population[Group] -
185  init.segment(first_index + LctStateGroup::template get_first_index<InfectionState::Exposed>(),
186  LctStateGroup::Count - 2)
187  .sum() -
188  deaths[Group];
189 
190  // Dead.
191  init[first_index + LctStateGroup::template get_first_index<InfectionState::Dead>()] = deaths[Group];
192 
193  for (size_t state_idx : {LctStateGroup::template get_first_index<InfectionState::Susceptible>(),
194  LctStateGroup::template get_first_index<InfectionState::Recovered>(),
195  LctStateGroup::template get_first_index<InfectionState::Dead>()}) {
196  if (floating_point_less(init[first_index + state_idx], 0.0, 1e-8)) {
197  log_error("Initialization failed. Values of total_confirmed_cases, deaths and total_population do not "
198  "match the specified values for the transitions, leading to at least one negative result.");
199  return true;
200  }
201  }
202  // Function call for next group if applicable.
203  if constexpr (Group + 1 < Model::num_groups) {
204  return compute_initialization_vector_impl<Group + 1>(init, total_population, deaths, total_confirmed_cases);
205  }
206  else {
207  return false;
208  }
209  }

215  bool check_constraints() const
216  {
217  if (!((Eigen::Index)InfectionTransition::Count * Model::num_groups == m_flows.get_num_elements())) {
218  log_error("Initial condition size does not match subcompartments.");
219  return true;
220  }
221 
222  if (!(floating_point_equal(0., m_flows.get_last_time(), 1e-8, 1e-14))) {
223  log_error("Last time point in flows has to be 0.");
224  return true;
225  }
226 
227  for (int i = 1; i < m_flows.get_num_time_points(); i++) {
228  if (!(floating_point_equal(m_dt, m_flows.get_time(i) - m_flows.get_time(i - 1), 1e-8, 1e-14))) {
229  log_error("Time points in flows have to be equidistant.");
230  return true;
231  }
232  }
233 
234  if (!(m_dt < 1)) {
235  log_warning("Step size was set very large. The result could be distorted.");
236  return true;
237  }
238 
239  // Check if model is valid and the calculated initial value vector is valid.
240  return m_model.check_constraints();
241  }

258  template <InfectionState State, size_t Group>
259  bool compute_compartment(Eigen::VectorX<ScalarType>& init, Eigen::Index idx_incoming_flow,
260  ScalarType transition_rate) const
261  {
262  size_t first_index = m_model.populations.template get_first_index_of_group<Group>() +
263  type_at_index_t<Group, LctStatesGroups>::template get_first_index<State>();
264  size_t num_subcompartments = type_at_index_t<Group, LctStatesGroups>::template get_num_subcompartments<State>();
265 
266  // Initialize relevant density for the InfectionState.
267  // For the first subcompartment a shape parameter of one is needed.
268  ErlangDensity erlang(1, 1. / (num_subcompartments * transition_rate));
269 
270  // Initialize other relevant parameters.
271  ScalarType calc_time{0};
272  Eigen::Index calc_time_index{0};
273  ScalarType state_age{0};
274  ScalarType sum{0};
275  Eigen::Index num_time_points = m_flows.get_num_time_points();
276 
277  // Calculate number of people in each subcompartment.
278  for (size_t j = 0; j < num_subcompartments; j++) {
279  // For subcompartment number j+1, shape parameter j+1 is needed.
280  erlang.set_distribution_parameter((ScalarType)j + 1);
281 
282  // Determine relevant calculation area and corresponding index.
283  calc_time = erlang.get_support_max(m_dt, m_tol);
284  calc_time_index = (Eigen::Index)std::ceil(calc_time / m_dt) - 1;
285 
286  if (num_time_points < calc_time_index) {
287  log_error("Initialization failed. Not enough time points for the transitions are given. More than {} "
288  "are needed but just {} are given.",
289  calc_time_index, num_time_points);
290  return true;
291  }
292  else {
293  // Approximate integral with non-standard scheme.
294  for (Eigen::Index i = num_time_points - calc_time_index; i < num_time_points; i++) {
295  state_age = (num_time_points - i) * m_dt;
296  sum += erlang.eval(state_age) * m_flows[i][idx_incoming_flow];
297  }
298  init[first_index + j] = 1. / (num_subcompartments * transition_rate) * sum;
299  if (init[first_index + j] < 0) {
300  log_error(
301  "Initialization failed. Result for at least one subcompartment is less than zero. Please check "
302  "the input values.");
303  return true;
304  }
305  }
306  sum = 0;
307  }
308  return false;
309  }
