81  Eigen::Ref<Eigen::VectorXd> advance(ScalarType tmax)
82  {
83  update_current_rates_and_waiting_times();
84  size_t next_event = determine_next_event(); // index of the next event
85  ScalarType current_time = m_result.get_last_time();
86  // set in the past to add a new time point immediately
87  ScalarType last_result_time = current_time - m_dt;
88  // iterate over time
89  while (current_time + m_waiting_times[next_event] < tmax) {
90  // update time
91  current_time += m_waiting_times[next_event];
92  // regularily save current state in m_results
93  if (current_time > last_result_time + m_dt) {
94  last_result_time = current_time;
95  m_result.add_time_point(current_time);
96  // copy from the previous last value
97  m_result.get_last_value() = m_result[m_result.get_num_time_points() - 2];
98  }
99  // decide event type by index and perform it
100  if (next_event < adoption_rates().size()) {
101  // perform adoption event
102  const auto& rate = adoption_rates()[next_event];
103  m_result.get_last_value()[m_model->populations.get_flat_index({rate.region, rate.from})] -= 1;
104  m_model->populations[{rate.region, rate.from}] -= 1;
105  m_result.get_last_value()[m_model->populations.get_flat_index({rate.region, rate.to})] += 1;
106  m_model->populations[{rate.region, rate.to}] += 1;
107  }
108  else {
109  // perform transition event
110  const auto& rate = transition_rates()[next_event - adoption_rates().size()];
111  m_result.get_last_value()[m_model->populations.get_flat_index({rate.from, rate.status})] -= 1;
112  m_model->populations[{rate.from, rate.status}] -= 1;
113  m_result.get_last_value()[m_model->populations.get_flat_index({rate.to, rate.status})] += 1;
114  m_model->populations[{rate.to, rate.status}] += 1;
115  }
116  // update internal times
117  for (size_t i = 0; i < m_internal_time.size(); i++) {
118  m_internal_time[i] += m_current_rates[i] * m_waiting_times[next_event];
119  }
120  // draw new "next event" time for the occured event
121  m_tp_next_event[next_event] +=
122  mio::ExponentialDistribution<ScalarType>::get_instance()(m_model->get_rng(), 1.0);
123  // precalculate next event
124  update_current_rates_and_waiting_times();
125  next_event = determine_next_event();
126  }
127  // copy last result, if no event occurs between last_result_time and tmax
128  if (last_result_time < tmax) {
129  m_result.add_time_point(tmax);
130  m_result.get_last_value() = m_result[m_result.get_num_time_points() - 2];
131  }
132  return m_result.get_last_value();
133  }

180  inline void update_current_rates_and_waiting_times()
181  {
182  size_t i = 0; // shared index for iterating both rates
183  for (const auto& rate : adoption_rates()) {
184  m_current_rates[i] = m_model->evaluate(rate, m_result.get_last_value());
185  m_waiting_times[i] = (m_current_rates[i] > 0)
186  ? (m_tp_next_event[i] - m_internal_time[i]) / m_current_rates[i]
187  : std::numeric_limits<ScalarType>::max();
188  i++;
189  }
190  for (const auto& rate : transition_rates()) {
191  m_current_rates[i] = m_model->evaluate(rate, m_result.get_last_value());
192  m_waiting_times[i] = (m_current_rates[i] > 0)
193  ? (m_tp_next_event[i] - m_internal_time[i]) / m_current_rates[i]
194  : std::numeric_limits<ScalarType>::max();
195  i++;
196  }
