81  bool check_constraints() const
82  {
83  auto params = this->parameters;
84 
85  // --- Check that the dimensions are consistent. ---
86  if ((Eigen::Index)LctState::template get_num_subcompartments<InfectionState::Exposed>() !=
87  params.template get<StartingProbabilitiesExposed>().rows()) {
88  log_error("Constraint check: Dimension of the parameters does not match the number of subcompartments for "
89  "the Exposed "
90  "compartment.");
91  return true;
92  }
93  if ((Eigen::Index)LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>() !=
94  params.template get<StartingProbabilitiesInfectedNoSymptoms>().rows()) {
95  log_error(
96  "Constraint check: Dimension of the parameters does not match the number of subcompartments for the "
97  "InfectedNoSymptoms compartment.");
98  return true;
99  }
100  if ((Eigen::Index)LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>() !=
101  params.template get<StartingProbabilitiesInfectedSymptoms>().rows()) {
102  log_error(
103  "Constraint check: Dimension of the parameters does not match the number of subcompartments for the "
104  "InfectedSymptoms compartment.");
105  return true;
106  }
107  if ((Eigen::Index)LctState::template get_num_subcompartments<InfectionState::InfectedSevere>() !=
108  params.template get<StartingProbabilitiesInfectedSevere>().rows()) {
109  log_error("Constraint check: Dimension of the parameters does not match the number of subcompartments for "
110  "the InfectedSevere "
111  "compartment.");
112  return true;
113  }
114  if ((Eigen::Index)LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() !=
115  params.template get<StartingProbabilitiesInfectedCritical>().rows()) {
116  log_error(
117  "Constraint check: Dimension of the parameters does not match the number of subcompartments for the "
118  "InfectedCritical compartment.");
119  return true;
120  }
121 
122  return (params.check_constraints() || this->populations.check_constraints());
123  }

137  void get_derivatives(Eigen::Ref<const Eigen::VectorX<ScalarType>> pop,
138  Eigen::Ref<const Eigen::VectorX<ScalarType>> y, ScalarType t,
139  Eigen::Ref<Eigen::VectorX<ScalarType>> dydt) const override
140  {
141  dydt.setZero();
142 
143  auto params = this->parameters;
144  auto total_population = pop.sum() - pop[LctState::template get_first_index<InfectionState::Dead>()];
145 
146  // Calculate sum of all subcompartments for InfectedNoSymptoms.
147  ScalarType InfectedNoSymptoms_sum =
148  pop.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
149  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
150  .sum();
151  // Calculate sum of all subcompartments for InfectedSymptoms.
152  ScalarType InfectedSymptoms_sum =
153  pop.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
154  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
155  .sum();
156 
157  // --- Susceptibles. ---
158  ScalarType season_val = 1 + params.template get<Seasonality>() *
159  sin(3.141592653589793 * ((params.template get<StartDay>() + t) / 182.5 + 0.5));
160  dydt[0] = -y[0] / total_population * season_val * params.template get<TransmissionProbabilityOnContact>() *
161  params.template get<ContactPatterns>().get_cont_freq_mat().get_matrix_at(t)(0, 0) *
162  (params.template get<RelativeTransmissionNoSymptoms>() * InfectedNoSymptoms_sum +
163  params.template get<RiskOfInfectionFromSymptomatic>() * InfectedSymptoms_sum);
164 
165  // --- Exposed. ---
166  dydt.segment(LctState::template get_first_index<InfectionState::Exposed>(),
167  LctState::template get_num_subcompartments<InfectionState::Exposed>()) -=
168  dydt[0] * params.template get<StartingProbabilitiesExposed>();
169  dydt.segment(LctState::template get_first_index<InfectionState::Exposed>(),
170  LctState::template get_num_subcompartments<InfectionState::Exposed>()) +=
171  params.template get<TransitionMatrixExposedToInfectedNoSymptoms>().transpose() *
172  y.segment(LctState::template get_first_index<InfectionState::Exposed>(),
173  LctState::template get_num_subcompartments<InfectionState::Exposed>());
174 
175  // --- InfectedNoSymptoms. ---
176  // Flow from Exposed To InfectedNoSymptoms.
177  dydt.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
178  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>()) =
179  -(params.template get<TransitionMatrixExposedToInfectedNoSymptoms>() *
180  Eigen::VectorX<ScalarType>::Ones(LctState::template get_num_subcompartments<InfectionState::Exposed>()))
181  .transpose() *
182  y.segment(LctState::template get_first_index<InfectionState::Exposed>(),
183  LctState::template get_num_subcompartments<InfectionState::Exposed>()) *
184  params.template get<StartingProbabilitiesInfectedNoSymptoms>();
185  // Flow from InfectedNoSymptoms To InfectedSymptoms.
186  size_t dimensionInfectedNoSymptomsToInfectedSymptoms =
187  params.template get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().rows();
188  dydt.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
189  dimensionInfectedNoSymptomsToInfectedSymptoms) +=
190  params.template get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>().transpose() *
191  y.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
192  dimensionInfectedNoSymptomsToInfectedSymptoms);
193  // Flow from InfectedNoSymptoms To Recovered.
194  size_t dimensionInfectedNoSymptomsToRecovered =
195  params.template get<TransitionMatrixInfectedNoSymptomsToRecovered>().rows();
196  dydt.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() +
197  dimensionInfectedNoSymptomsToInfectedSymptoms,
198  dimensionInfectedNoSymptomsToRecovered) +=
199  params.template get<TransitionMatrixInfectedNoSymptomsToRecovered>().transpose() *
200  y.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() +
201  dimensionInfectedNoSymptomsToInfectedSymptoms,
202  dimensionInfectedNoSymptomsToRecovered);
203  // Add flow directly to Recovered compartment.
204  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +=
205  -(params.template get<TransitionMatrixInfectedNoSymptomsToRecovered>() *
206  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedNoSymptomsToRecovered))
207  .transpose() *
208  y.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() +
209  dimensionInfectedNoSymptomsToInfectedSymptoms,
210  dimensionInfectedNoSymptomsToRecovered);
211 
212  // --- InfectedSymptoms. ---
213  // Flow from InfectedNoSymptoms To InfectedSymptoms.
214  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
215  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>()) =
216  -params.template get<StartingProbabilitiesInfectedSymptoms>() *
217  (params.template get<TransitionMatrixInfectedNoSymptomsToInfectedSymptoms>() *
218  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedNoSymptomsToInfectedSymptoms))
219  .transpose() *
220  y.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
221  dimensionInfectedNoSymptomsToInfectedSymptoms);
222  // Flow from InfectedSymptoms To InfectedSevere.
223  size_t dimensionInfectedSymptomsToInfectedSevere =
224  params.template get<TransitionMatrixInfectedSymptomsToInfectedSevere>().rows();
225  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
226  dimensionInfectedSymptomsToInfectedSevere) +=
227  params.template get<TransitionMatrixInfectedSymptomsToInfectedSevere>().transpose() *
228  y.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
229  dimensionInfectedSymptomsToInfectedSevere);
230  // Flow from InfectedSymptoms To Recovered.
231  size_t dimensionInfectedSymptomsToRecovered =
232  params.template get<TransitionMatrixInfectedSymptomsToRecovered>().rows();
233  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>() +
234  dimensionInfectedSymptomsToInfectedSevere,
235  dimensionInfectedSymptomsToRecovered) +=
236  params.template get<TransitionMatrixInfectedSymptomsToRecovered>().transpose() *
237  y.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>() +
238  dimensionInfectedSymptomsToInfectedSevere,
239  dimensionInfectedSymptomsToRecovered);
240  // Add flow directly to Recovered compartment.
241  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +=
242  -(params.template get<TransitionMatrixInfectedSymptomsToRecovered>() *
243  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedSymptomsToRecovered))
244  .transpose() *
245  y.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>() +
246  dimensionInfectedSymptomsToInfectedSevere,
247  dimensionInfectedSymptomsToRecovered);
248 
249  // --- InfectedSevere. ---
250  // Flow from InfectedSymptoms To InfectedSevere.
251  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
252  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>()) =
253  -params.template get<StartingProbabilitiesInfectedSevere>() *
254  (params.template get<TransitionMatrixInfectedSymptomsToInfectedSevere>() *
255  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedSymptomsToInfectedSevere))
256  .transpose() *
257  y.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
258  dimensionInfectedSymptomsToInfectedSevere);
259  // Flow from InfectedSevere To InfectedCritical.
260  size_t dimensionInfectedSevereToInfectedCritical =
261  params.template get<TransitionMatrixInfectedSevereToInfectedCritical>().rows();
262  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
263  dimensionInfectedSevereToInfectedCritical) +=
264  params.template get<TransitionMatrixInfectedSevereToInfectedCritical>().transpose() *
265  y.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
266  dimensionInfectedSevereToInfectedCritical);
267  // Flow from InfectedSevere To Recovered.
268  size_t dimensionInfectedSevereToRecovered =
269  params.template get<TransitionMatrixInfectedSevereToRecovered>().rows();
270  dydt.segment(LctState::template get_first_index<InfectionState::InfectedSevere>() +
271  dimensionInfectedSevereToInfectedCritical,
272  dimensionInfectedSevereToRecovered) +=
273  params.template get<TransitionMatrixInfectedSevereToRecovered>().transpose() *
274  y.segment(LctState::template get_first_index<InfectionState::InfectedSevere>() +
275  dimensionInfectedSevereToInfectedCritical,
276  dimensionInfectedSevereToRecovered);
277  // Add flow directly to Recovered compartment.
278  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +=
279  -(params.template get<TransitionMatrixInfectedSevereToRecovered>() *
280  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedSevereToRecovered))
281  .transpose() *
282  y.segment(LctState::template get_first_index<InfectionState::InfectedSevere>() +
283  dimensionInfectedSevereToInfectedCritical,
284  dimensionInfectedSevereToRecovered);
285 
286  // --- InfectedCritical. ---
287  // Flow from InfectedSevere To InfectedCritical.
288  dydt.segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
289  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>()) =
290  -params.template get<StartingProbabilitiesInfectedCritical>() *
291  (params.template get<TransitionMatrixInfectedSevereToInfectedCritical>() *
292  Eigen::VectorX<ScalarType>::Ones(dimensionInfectedSevereToInfectedCritical))
293  .transpose() *
294  y.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
295  dimensionInfectedSevereToInfectedCritical);
296  // Flow from InfectedCritical To Dead.
297  size_t dimensionInfectedCriticalToDead = params.template get<TransitionMatrixInfectedCriticalToDead>().rows();
298  dydt.segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
299  dimensionInfectedCriticalToDead) +=
300  params.template get<TransitionMatrixInfectedCriticalToDead>().transpose() *
301  y.segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
302  dimensionInfectedCriticalToDead);
303  // Flow from InfectedCritical To Recovered.
304  size_t dimensionInfectedCriticalToRecovered =
305  params.template get<TransitionMatrixInfectedCriticalToRecovered>().rows();
306  dydt.segment(LctState::template get_first_index<InfectionState::InfectedCritical>() +
307  dimensionInfectedCriticalToDead,
308  dimensionInfectedCriticalToRecovered) +=
309  params.template get<TransitionMatrixInfectedCriticalToRecovered>().transpose() *
310  y.segment(LctState::template get_first_index<InfectionState::InfectedCritical>() +
311  dimensionInfectedCriticalToDead,
312  dimensionInfectedCriticalToRecovered);
313  // Add flow directly to Recovered compartment.

343  TimeSeries<ScalarType> calculate_compartments(const TimeSeries<ScalarType>& subcompartments_ts) const
344  {
345  TimeSeries<ScalarType> compartments_ts((Eigen::Index)InfectionState::Count);
346  if (!(LctState::Count == subcompartments_ts.get_num_elements())) {
347  log_error("Result does not match InfectionStates of the model.");
348  // Return a TimeSeries with values -1.
349  Eigen::VectorX<ScalarType> error_output =
350  Eigen::VectorX<ScalarType>::Constant((Eigen::Index)InfectionState::Count, -1);
351  compartments_ts.add_time_point(-1, error_output);
352  return compartments_ts;
353  }
354  Eigen::VectorX<ScalarType> compartments((Eigen::Index)InfectionState::Count);
355  for (Eigen::Index i = 0; i < subcompartments_ts.get_num_time_points(); ++i) {
356  compartments_ts.add_time_point(subcompartments_ts.get_time(i),
357  LctState::calculate_compartments(subcompartments_ts[i]));
358  }
359  return compartments_ts;
360  }
