160  Model model{params_};

527  // Change the Location of a Person. this requires that Location is part of this Model.
535  inline void change_location(Person& person, LocationId destination, TransportMode mode = TransportMode::Unknown,
536  const std::vector<uint32_t>& cells = {0})
537  {
538  LocationId origin = get_location(person).get_id();
539  const bool has_changed_location = mio::abm::change_location(person, get_location(destination), mode, cells);
540  // if the person has changed location, invalidate exposure caches but keep population caches valid
541  if (has_changed_location) {
542  m_are_exposure_caches_valid = false;
543  if (m_is_local_population_cache_valid) {
544  --m_local_population_cache[origin.get()];
545  ++m_local_population_cache[destination.get()];
546  }
547  }
548  }

585  inline void interact(Person& person, TimePoint t, TimeSpan dt)
586  {
587  if (!m_are_exposure_caches_valid) {
588  // checking caches is only needed for external calls
589  // during simulation (i.e. in evolve()), the caches are computed in begin_step
590  compute_exposure_caches(t, dt);
591  m_are_exposure_caches_valid = true;
592  }
593  auto personal_rng = PersonalRandomNumberGenerator(person);
594  mio::abm::interact(personal_rng, person, get_location(person.get_location()),
595  m_air_exposure_rates_cache[person.get_location().get()],
596  m_contact_exposure_rates_cache[person.get_location().get()], t, dt, parameters);
597  }

606  template <class M>
607  static std::conditional_t<std::is_const_v<M>, const Person&, Person&> get_person_impl(M& m, PersonId person_id)
608  {
609  if (m.m_person_ids_equal_index) {
610  assert(static_cast<uint32_t>(person_id.get()) < m.m_persons.size() &&
611  "Given PersonId is not in this Model.");
612  return m.m_persons[static_cast<uint32_t>(person_id.get())];
613  }
614  else {
615  mio::log_warning("get_person is accessed by PersonId which does not align with the index of the person due "
616  "to former removal of persons. Therefore m_persons is searched.");
617  auto it = std::find_if(m.m_persons.begin(), m.m_persons.end(), [person_id](auto& person) {
618  return person.get_id() == person_id;
619  });
620  assert(it != m.m_persons.end() && "Given PersonId is not in this Model.");
621  return *it;
622  };
623  }
