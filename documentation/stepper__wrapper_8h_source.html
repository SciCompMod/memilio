80  bool step(const mio::DerivFunction<FP>& f, Eigen::Ref<Eigen::VectorX<FP> const> yt, FP& t, FP& dt,
81  Eigen::Ref<Eigen::VectorX<FP>> ytp1) const override
82  {
83  using boost::numeric::odeint::fail;
84  using std::max;
85  assert(0 <= this->get_dt_min());
86  assert(this->get_dt_min() <= this->get_dt_max());
87 
88  if (dt < this->get_dt_min() || dt > this->get_dt_max()) {
89  mio::log_warning("IntegratorCore: Restricting given step size dt = {} to [{}, {}].", dt, this->get_dt_min(),
90  this->get_dt_max());
91  }
92  // set initial values for exit conditions
93  auto step_result = fail;
94  bool is_dt_valid = true;
95  // copy vectors from the references, since the stepper cannot (trivially) handle Eigen::Ref
96  m_ytp1 = ytp1; // y(t')
97  m_yt = yt; // y(t)
98  // make a integration step, adapting dt to a possibly larger value on success,
99  // or a strictly smaller value on fail.
100  // stop only on a successful step or a failed step size adaption (w.r.t. the minimal step size dt_min)
101  while (step_result == fail && is_dt_valid) {
102  if (dt < this->get_dt_min()) {
103  is_dt_valid = false;
104  dt = this->get_dt_min();
105  }
106  // we use the scheme try_step(sys, in, t, out, dt) with sys=f, in=y(t), out=y(t').
107  // this is similiar to do_step, but it can adapt the step size dt. If successful, it also updates t.
108  // Note: the resizer used by m_stepper restricts dt to dt_max (via making a failed step)
109  if constexpr (!is_fsal_stepper) { // prevent compile time errors with fsal steppers
110  step_result = m_stepper.try_step(
111  // reorder arguments of the DerivFunction f for the stepper
112  [&](const Eigen::VectorX<FP>& x, Eigen::VectorX<FP>& dxds, FP s) {
113  f(x, s, dxds);
114  },
115  m_yt, t, m_ytp1, dt);
116  }
117  }
118  // output the new value by copying it back to the reference
119  ytp1 = m_ytp1;
120  // bound dt from below
121  // the last adaptive step (successful or not) may have calculated a new step size smaller than m_dt_min
122 
123  dt = max(dt, this->get_dt_min());
124  // check whether the last step failed (which means that m_dt_min was still too large to suffice tolerances)
125  if (step_result == fail) {
126  // adaptive stepping failed, but we still return the result of the last attempt
127  t += this->get_dt_min();
128  return false;
129  }
130  else {
131  // successfully made an integration step
132  return true;
133  }
134  }

211  // copy the values from y(t) to ytp1, since we use the scheme do_step(sys, inout, t, dt) with
212  // sys=f, inout=y(t) for in-place computation - also, this form is shared by several steppers in boost
213  ytp1 = yt;

215  // reorder arguments of the DerivFunction f for the stepper

217  f(x, s, dxds);
218  },
219  ytp1, t, dt);
220  // update time (it is not modified by do_step)
221  t += dt;
222  return true; // no step size adaption
223  }
