87 template <class Graph, class Timepoint = double, class Timespan = double,
88  class edge_f = void (*)(Timepoint, Timespan, typename Graph::EdgeProperty&, typename Graph::NodeProperty&,
89  typename Graph::NodeProperty&),
90  class node_f = void (*)(Timepoint, Timespan, typename Graph::NodeProperty&)>
91 class GraphSimulation : public GraphSimulationBase<Graph, Timepoint, Timespan, edge_f, node_f>
92 {
93  using Base = GraphSimulationBase<Graph, Timepoint, Timespan, edge_f, node_f>;
94  using Base::GraphSimulationBase;
95 
96 public:
97  void advance(Timepoint t_max = 1.0)
98  {
99  auto dt = Base::m_dt;
100  while (Base::m_t < t_max) {
101  if (Base::m_t + dt > t_max) {
102  dt = t_max - Base::m_t;
103  }
104 
105  for (auto& n : Base::m_graph.nodes()) {
106  Base::m_node_func(Base::m_t, dt, n.property);
107  }
108 
109  Base::m_t += dt;
110 
111  for (auto& e : Base::m_graph.edges()) {
112  Base::m_edge_func(Base::m_t, dt, e.property, Base::m_graph.nodes()[e.start_node_idx].property,
113  Base::m_graph.nodes()[e.end_node_idx].property);
114  }
115  }
116  }
117 };

120 class GraphSimulationStochastic
121  : public GraphSimulationBase<Graph, double, double,
122  std::function<void(typename Graph::EdgeProperty&, size_t,
123  typename Graph::NodeProperty&, typename Graph::NodeProperty&)>,
124  std::function<void(double, double, typename Graph::NodeProperty&)>>

150  void advance(double t_max)
151  {
152  //draw normalized waiting time
153  ScalarType normalized_waiting_time = ExponentialDistribution<ScalarType>::get_instance()(m_rng, 1.0);
154  std::vector<ScalarType> dt_cand(Base::m_graph.nodes().size());
155  ScalarType cumulative_rate = 0; //cumulative transition rate
156  size_t parameters_per_edge =
157  size_t(Base::m_graph.edges()[0].property.get_parameters().get_coefficients().get_shape().rows());
158  std::vector<ScalarType> transition_rates(parameters_per_edge * Base::m_graph.edges().size());
159  while (Base::m_t < t_max) {
160  Base::m_dt = std::min({Base::m_dt, t_max - Base::m_t});
161  //calculate current transition rates and cumulative rate
162  cumulative_rate = get_cumulative_transition_rate();
163  if (cumulative_rate * Base::m_dt >
164  normalized_waiting_time) { //at least one transition event during current time step
165  do {
166  //evaluate rates
167  get_rates(m_rates);
168  //draw transition event
169  size_t event = mio::DiscreteDistribution<size_t>::get_instance()(m_rng, m_rates);
170  //edge that performs transition event
171  auto& event_edge = Base::m_graph.edges()[event / parameters_per_edge];
172  //index for compartment and age group moving
173  auto flat_index = event % parameters_per_edge;
174 
175  //advance nodes until t + (waiting_time / cumulative_rate)
176  for (size_t node_iter = 0; node_iter < Base::m_graph.nodes().size(); ++node_iter) {
177  auto& node = Base::m_graph.nodes()[node_iter];
178  Base::m_node_func(Base::m_t, normalized_waiting_time / cumulative_rate, node.property);
179  }
180 
181  //advance time
182  Base::m_t += normalized_waiting_time / cumulative_rate;
183 
184  //reduce remaining time of current time step
185  Base::m_dt -= normalized_waiting_time / cumulative_rate;
186 
187  //perform transition
188  Base::m_edge_func(event_edge.property, flat_index,
189  Base::m_graph.nodes()[event_edge.start_node_idx].property,
190  Base::m_graph.nodes()[event_edge.end_node_idx].property);
191 
192  //calculate new cumulative rate
193  cumulative_rate = get_cumulative_transition_rate();
194 
195  //draw new normalized waiting time
196  normalized_waiting_time = ExponentialDistribution<ScalarType>::get_instance()(m_rng, 1.0);
197 
198  } while (cumulative_rate * Base::m_dt > normalized_waiting_time);
199  }
200  else { //no transition event in current time step
201  normalized_waiting_time -= cumulative_rate * Base::m_dt; //reduce waiting time by current time step
202  }
203 
204  //advance nodes until t+dt
205  for (size_t node_iter = 0; node_iter < Base::m_graph.nodes().size(); ++node_iter) {
206  auto& node = Base::m_graph.nodes()[node_iter];
207  Base::m_node_func(Base::m_t, Base::m_dt, node.property);
208  //get new dt of each node
209  dt_cand[node_iter] = node.property.get_simulation().get_dt();
210  }
211 
212  //advance time
213  Base::m_t += Base::m_dt;
214  //new dt ist the minimal dt of all nodes
215  Base::m_dt = *std::min_element(dt_cand.begin(), dt_cand.end());
216  }
217  }
218 
219  RandomNumberGenerator& get_rng()
220  {
221  return m_rng;
222  }
223 
224 private:
225  ScalarType get_cumulative_transition_rate()
226  {
227  //compute current cumulative transition rate
228  ScalarType cumulative_transition_rate = 0;
229  for (auto& e : Base::m_graph.edges()) {
230  cumulative_transition_rate +=
231  e.property.get_transition_rates(Base::m_graph.nodes()[e.start_node_idx].property).sum();
232  }
233  return cumulative_transition_rate;
234  }
235 
236  void get_rates(std::vector<ScalarType>& rates)
237  {
238  size_t j = 0;
239  for (auto& e : Base::m_graph.edges()) {
240  auto edge_rates = e.property.get_transition_rates(Base::m_graph.nodes()[e.start_node_idx].property);
241  for (Eigen::Index i = 0; i < edge_rates.size(); ++i) {
242  const auto compartment_value =
243  Base::m_graph.nodes()[e.start_node_idx].property.get_result().get_last_value()[i];
244  rates[j] = (compartment_value < 1.) ? 0. : edge_rates(i);
245 
246  j++;
247  }
248  }
249  }
